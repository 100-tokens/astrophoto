**Tech Stack:** Rust (Axum + sqlx 0.8 + testcontainers), Postgres 16, ts-rs, SvelteKit 2 (Svelte 5 runes), Playwright.

-- 0018 equipment catalog enriched
--
-- 1. Catalog metadata on equipment_items.
alter table equipment_items
    add column status        text        not null default 'approved'
        check (status in ('pending','approved','rejected','merged')),
    add column submitted_by  uuid        references users(id) on delete set null,
    add column approved_at   timestamptz,
    add column created_at    timestamptz not null default now();

update equipment_items
   set approved_at = coalesce(approved_at, now())
 where status = 'approved';

create index equipment_items_status_idx
    on equipment_items (kind, status, usage_count desc)
    where status = 'approved';

-- 2. telescope_specs.
create table telescope_specs (
    item_id         uuid primary key references equipment_items(id) on delete cascade,
    design          text       check (design in (
                       'refractor_apo','refractor_achro','sct','rc',
                       'newtonian','maksutov_cassegrain','maksutov_newtonian',
                       'dall_kirkham','other')),
    aperture_mm     int        check (aperture_mm between 30 and 1500),
    focal_length_mm int        check (focal_length_mm between 100 and 15000),
    focal_ratio_f   numeric(4,2) generated always as
                        ((focal_length_mm::numeric) / nullif(aperture_mm, 0)) stored
);
create index telescope_specs_aperture_idx on telescope_specs (aperture_mm);
create index telescope_specs_focal_idx    on telescope_specs (focal_length_mm);

-- 3. camera_specs.
create table camera_specs (
    item_id           uuid primary key references equipment_items(id) on delete cascade,
    sensor_type       text  check (sensor_type in ('cmos','ccd')),
    color_type        text  check (color_type in ('mono','osc')),
    cooled            boolean,
    sensor_model      text,
    pixel_size_um     numeric(4,2) check (pixel_size_um between 0.5 and 25),
    sensor_width_px   int   check (sensor_width_px > 0),
    sensor_height_px  int   check (sensor_height_px > 0)
);

-- 4. filter_specs.
create table filter_specs (
    item_id        uuid primary key references equipment_items(id) on delete cascade,
    filter_type    text  check (filter_type in (
                      'luminance','red','green','blue',
                      'h_alpha','oiii','sii','uv_ir_cut',
                      'dual_band','tri_band','quad_band',
                      'light_pollution','broadband_color','other')),
    bandwidth_nm   numeric(5,2) check (bandwidth_nm > 0 and bandwidth_nm <= 200),
    size           text  check (size in (
                      '1_25in','2in','31mm','36mm','50mm_round','50mm_square','other')),
    mounted        boolean
);
create index filter_specs_type_idx on filter_specs (filter_type);

-- 5. mount_specs.
create table mount_specs (
    item_id     uuid primary key references equipment_items(id) on delete cascade,
    mount_type  text  check (mount_type in (
                   'equatorial_german','equatorial_fork','alt_az',
                   'harmonic_drive','strain_wave','other')),
    payload_kg  numeric(4,1) check (payload_kg > 0 and payload_kg <= 200),
    goto        boolean
);

-- 6. focal_modifier_specs.
create table focal_modifier_specs (
    item_id        uuid primary key references equipment_items(id) on delete cascade,
    modifier_type  text  check (modifier_type in (
                      'reducer','flattener','reducer_flattener',
                      'barlow','extender','corrector')),
    factor         numeric(3,2) check (factor > 0 and factor <= 5)
);

-- 7. photo_filters junction.
create table photo_filters (
    photo_id  uuid     not null references photos(id) on delete cascade,
    item_id   uuid     not null references equipment_items(id) on delete restrict,
    position  smallint not null default 0,
    primary key (photo_id, item_id)
);
create index photo_filters_item_idx      on photo_filters (item_id);
create index photo_filters_photo_pos_idx on photo_filters (photo_id, position);

//! Per-kind validation for EquipmentSpecsPayload. The DB check
//! constraints are the hard rules — this layer only catches
//! mismatched (kind, payload) pairs before we hit them.

use crate::api_types::EquipmentSpecsPayload;
use crate::error::AppError;

pub fn ensure_matches_kind(kind: &str, payload: &EquipmentSpecsPayload) -> Result<(), AppError> {
    let ok = match (kind, payload) {
        ("telescope",      EquipmentSpecsPayload::Telescope(_))      => true,
        ("camera",         EquipmentSpecsPayload::Camera(_))         => true,
        ("filter",         EquipmentSpecsPayload::Filter(_))         => true,
        ("mount",          EquipmentSpecsPayload::Mount(_))          => true,
        ("focal_modifier", EquipmentSpecsPayload::FocalModifier(_))  => true,

use crate::auth::middleware::CurrentUser;

    State(state): State<AppState>,
    _user: CurrentUser,
    Path(id): Path<Uuid>,
    Json(input): Json<EquipmentItemPatch>,
) -> Result<impl IntoResponse, AppError> {
    let mut tx = state.pool.begin().await?;
    let item = sqlx::query!("select kind from equipment_items where id=$1 for update", id)
        .fetch_optional(&mut *tx).await?
        .ok_or_else(|| AppError::NotFound("equipment item not found".into()))?;

        let trimmed = new_name.trim();
        if trimmed.is_empty() {
            return Err(AppError::Validation("display_name cannot be empty".into()));
        }
        sqlx::query!(
            "update equipment_items
                set display_name = $1, canonical_name = $2
              where id = $3",
            trimmed, trimmed.to_lowercase(), id
        ).execute(&mut *tx).await?;
    }

    if let Some(payload) = input.specs.as_ref() {
        crate::equipment::specs::ensure_matches_kind(&item.kind, payload)?;
        replace_specs(&mut tx, id, &item.kind, payload).await?;

    crate::equipment::items_get::handler(State(state), _user, Path(id)).await
}

    // `table` is a controlled constant, NOT user input — safe to interpolate.

```svelte
<script lang="ts">
  import { onMount } from 'svelte';
  import FilterChip from './FilterChip.svelte';
  import { FILTER_TYPE_META, bandwidthLabel } from '$lib/equipment/filter-types';
  import type { PhotoFilterChip, FilterType } from '$lib/api/types';

  type AutocompleteRow = {
    id: string;
    display_name: string;
    filter_type: FilterType | null;
    bandwidth_nm: number | null;
    usage_count: number;
  };

  type Props = {
    value: PhotoFilterChip[];
    orphans?: string[];
    startOpen?: boolean;
    placeholder?: string;
    onChange: (next: PhotoFilterChip[]) => void;
  };

  let {
    value,
    orphans = [],
    startOpen = false,
    placeholder = 'Search filters…',
    onChange,
  }: Props = $props();

  let items = $state<PhotoFilterChip[]>([...value]);
  let query = $state('');
  let open = $state(startOpen);
  let focusIdx = $state(0);
  let dragId = $state<string | null>(null);
  let inputEl: HTMLInputElement | undefined;
  let matches = $state<AutocompleteRow[]>([]);
  let loading = $state(false);

  // Keep `items` in sync with the parent's `value` if it changes externally.
  $effect(() => {
    items = [...value];
  });

  async function fetchMatches(q: string) {
    loading = true;
    try {
      const params = new URLSearchParams({ kind: 'filter', q });
      const r = await fetch(`/api/equipment/autocomplete?${params}`, { credentials: 'include' });
      if (!r.ok) throw new Error('autocomplete failed');
      const rows: AutocompleteRow[] = await r.json();
      const selected = new Set(items.map((i) => i.id));
      matches = rows.filter((r) => !selected.has(r.id)).slice(0, 8);
    } catch {
      matches = [];
    } finally {
      loading = false;
    }
  }

  $effect(() => {
    if (open) fetchMatches(query);
  });

  function add(f: AutocompleteRow) {
    const next = [...items, {
      id: f.id,
      display_name: f.display_name,
      filter_type: f.filter_type,
      bandwidth_nm: f.bandwidth_nm,
      position: items.length,
    }];
    items = next;
    query = '';
    focusIdx = 0;
    onChange(next);
  }

  function remove(id: string) {
    const next = items
      .filter((f) => f.id !== id)
      .map((f, i) => ({ ...f, position: i }));
    items = next;
    onChange(next);
  }

  function reorder(srcId: string, beforeId: string | null) {
    if (srcId === beforeId) return;
    const src = items.find((f) => f.id === srcId);
    if (!src) return;
    const rest = items.filter((f) => f.id !== srcId);
    const idx = beforeId == null ? rest.length : rest.findIndex((f) => f.id === beforeId);
    const next = [...rest.slice(0, idx), src, ...rest.slice(idx)].map((f, i) => ({ ...f, position: i }));
    items = next;
    onChange(next);
  }

  async function createNew() {
    const trimmed = query.trim();
    if (!trimmed) return;
    const r = await fetch('/api/equipment/items', {
      method: 'POST',
      credentials: 'include',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ kind: 'filter', display_name: trimmed }),
    });
    if (!r.ok) return;
    const created = await r.json();
    add({
      id: created.id,
      display_name: created.display_name,
      filter_type: null,
      bandwidth_nm: null,
      usage_count: created.usage_count,
    });
  }

  function onKey(e: KeyboardEvent) {
    if (e.key === 'ArrowDown') {
      e.preventDefault();
      focusIdx = Math.min(matches.length - 1, focusIdx + 1);
    } else if (e.key === 'ArrowUp') {
      e.preventDefault();
      focusIdx = Math.max(0, focusIdx - 1);
    } else if (e.key === 'Enter') {
      e.preventDefault();
      if (matches[focusIdx]) add(matches[focusIdx]);
      else if (query.trim()) createNew();
    } else if (e.key === 'Backspace' && !query && items.length) {
      remove(items[items.length - 1].id);
    } else if (e.key === 'Escape') {
      open = false;
    }
  }
</script>

<div class="fchip-input" onclick={() => { inputEl?.focus(); open = true; }} role="presentation">
  {#each items as f (f.id)}
    <span
      draggable="true"
      ondragstart={(e) => { dragId = f.id; e.dataTransfer!.effectAllowed = 'move'; }}
      ondragover={(e) => e.preventDefault()}
      ondrop={(e) => { e.preventDefault(); if (dragId) { reorder(dragId, f.id); dragId = null; } }}
      ondragend={() => (dragId = null)}
      style="display:inline-flex"
    >
      <FilterChip filter={f} draggable removable dragging={dragId === f.id} onRemove={() => remove(f.id)} />
    </span>
  {/each}
  {#each orphans as tok (tok)}
    <span class="fchip-orphan" title="Legacy text filter — no catalog match yet">
      <span class="lbl">legacy</span>{tok}
    </span>
  {/each}
  <input
    bind:this={inputEl}
    bind:value={query}
    onfocus={() => (open = true)}
    onblur={() => setTimeout(() => (open = false), 150)}
    onkeydown={onKey}
    placeholder={items.length ? '' : placeholder}
  />
  {#if open}
    <div class="fchip-pop" onmousedown={(e) => e.preventDefault()} role="presentation">
      <div class="fchip-pop-head">
        <span>{query ? `MATCHES "${query}"` : 'POPULAR FILTERS'}</span>
        <span style="color: var(--fg-faint)">{matches.length}{loading ? ' · …' : ''}</span>
      </div>
      <div class="fchip-pop-list">
        {#if matches.length === 0 && !loading}
          <div style="padding: 14px; color: var(--fg-muted); font-size: 12px; font-family: var(--font-mono)">
            No matches. Press <span style="color: var(--accent)">↵ Enter</span> to create a new filter item.
          </div>
        {/if}
        {#each matches as f, i (f.id)}
          <button
            type="button"
            class={'fchip-pop-item' + (i === focusIdx ? ' is-focus' : '')}
            onclick={() => add(f)}
            onmouseenter={() => (focusIdx = i)}
          >
            <FilterChip filter={{ ...f, position: 0 }} compact />
            <span class="meta">
              {f.filter_type ? `${FILTER_TYPE_META[f.filter_type].label.toUpperCase()}` : 'UNTYPED'}
              {#if bandwidthLabel(f)} · {bandwidthLabel(f)}{/if}
            </span>
            <span class="usage">{f.usage_count.toLocaleString()} PHOTOS</span>
          </button>
        {/each}
      </div>
      {#if query.trim()}
        <button type="button" class="fchip-pop-create" onclick={createNew}>
          <svg width="12" height="12" viewBox="0 0 12 12" fill="none" stroke="currentColor" stroke-width="1.4">
            <path d="M6 2v8M2 6h8" stroke-linecap="round"/>
          </svg>
          <span>Create new · "<strong style="color: var(--fg-primary)">{query}</strong>"</span>
          <span class="kbd">↵ ENTER</span>
        </button>
      {/if}
    </div>
  {/if}
</div>
```
